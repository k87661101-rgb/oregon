            dragging = false
            if dragMoved < 8 then
                doToggle()
            end
        end
    end)

    return ToggleBtn

function spawnRainbowOrb(pos)
    pcall(function()
        -- Get the gun muzzle position
        local char = game.Players.LocalPlayer.Character
        local tool = char and char:FindFirstChildOfClass("Tool")
        local startPos
        if tool and tool:FindFirstChild("Handle") then
            local handle = tool.Handle
            local muzzle = handle:FindFirstChild("GunMuzzlePoint") or handle:FindFirstChild("GunFirePoint")
            if muzzle then
                startPos = muzzle.WorldCFrame.Position
            else
                startPos = handle.Position
            end
        else
            startPos = pos
        end

        -- Direction from gun to target
        local direction = (pos - startPos).Unit
        local orb = Instance.new("Part")
        orb.Anchored = false
        orb.CanCollide = false
        orb.CanQuery = false
        orb.CanTouch = false
        orb.Material = Enum.Material.Neon
        orb.Shape = Enum.PartType.Ball
        orb.Size = Vector3.new(4, 4, 4)
        orb.CFrame = CFrame.new(startPos)
        orb.Color = Color3.fromHSV(math.random(), 1, 1)
        orb.Parent = workspace
        -- Fly forward from the gun
        orb.AssemblyLinearVelocity = direction * 200
        game:GetService("Debris"):AddItem(orb, 1.5)
    end)
end

        local idx = table.find(teleportNames, name)
        if idx then table.remove(teleportNames, idx); table.insert(teleportNames, 1, name) end
    end
end

local smartTeleportKeywords = {
    ["💰 New Seller"] = {"new seller", "seller", "sell", "dealer"},
    ["🍴 New Deli"] = {"new deli", "deli"},
}

function getInstanceTeleportCFrame(inst)
    if not inst then return nil end
    local ok, cf = pcall(function()
        if inst:IsA("BasePart") then return inst.CFrame end
        if inst:IsA("Model") then return inst:GetPivot() end
        local part = inst:FindFirstChildWhichIsA("BasePart", true)
        return part and part.CFrame or nil
    end)
    if ok and cf then return cf + Vector3.new(0, 3, 0) end
    return nil
end

function resolveSmartTeleportCFrame(locationName)
    local stored = teleportLocations and teleportLocations[locationName]
    if typeof(stored) == "CFrame" then return stored end
    local keywords = smartTeleportKeywords[locationName]
    if not keywords then return nil end
    local best, bestScore = nil, -1
    local localChar = game:GetService("Players").LocalPlayer.Character
    for _, inst in ipairs(workspace:GetDescendants()) do
        if (not localChar or not inst:IsDescendantOf(localChar)) and (inst:IsA("BasePart") or inst:IsA("Model")) then
            local lname = string.lower(inst.Name or "")
            local score = 0
            for _, key in ipairs(keywords) do
                if string.find(lname, string.lower(key), 1, true) then score += #key end
            end
            if score > 0 and score > bestScore then
                local cf = getInstanceTeleportCFrame(inst)
                if cf then best, bestScore = cf, score end
            end
        end
    end
    return best
end

function teleportToLocationName(locationName)
    local cf = resolveSmartTeleportCFrame(locationName)
    if cf then
        Config:Teleport(cf)
        _fixNotify("TP", "Teleported to " .. tostring(locationName), 2)
        return true
    end
    _fixNotify("TP", "Could not find " .. tostring(locationName) .. " in this server.", 3)
    return false
end

function getPing()

function duplicateItem()
    if DupeCooldown then return end
    DupeCooldown = true

    local lp = game.Players.LocalPlayer
    local rs = game:GetService("ReplicatedStorage")
    local character = lp.Character
    local backpack = lp:FindFirstChild("Backpack")
    if not character or not backpack then
        DupeCooldown = false
        return
    end

    local tool = character:FindFirstChildOfClass("Tool")
    if not tool then
        DupeCooldown = false
        return
    end

    local toolName = tool.Name
    local toolId = nil

    -- Move tool to backpack
    tool.Parent = backpack
    task.wait(0.5)

    -- Set up listener for market item
    local connection
    if rs:FindFirstChild("MarketItems") then
        connection = rs.MarketItems.ChildAdded:Connect(function(item)
            if item.Name == toolName then
                local owner = item:WaitForChild("owner", 2)
                if owner and owner.Value == lp.Name then
                    toolId = item:GetAttribute("SpecialId")
                end
            end
        end)
    end

    -- List the weapon
    task.spawn(function()
        if rs:FindFirstChild("ListWeaponRemote") then
            rs.ListWeaponRemote:FireServer(toolName, 676767)
        end
    end)

    local ping = getPing()
    local delayTime = 0.25 + ((math.clamp(ping, 0, 300) / 300) * 0.03)
    task.wait(delayTime)

    -- Store in backpack
    task.spawn(function()
        if rs:FindFirstChild("BackpackRemote") then
            rs.BackpackRemote:InvokeServer("Store", toolName)
        end
    end)

    task.wait(3)

    -- Remove from market and grab back
    if toolId then
        task.spawn(function()
            if rs:FindFirstChild("BuyItemRemote") then
                rs.BuyItemRemote:FireServer(toolName, "Remove", toolId)
            end
        end)
    end

    task.spawn(function()
        if rs:FindFirstChild("BackpackRemote") then
            rs.BackpackRemote:InvokeServer("Grab", toolName)
        end
    end)

    task.wait(1)

    -- Re-equip
    local dupedTool = backpack:FindFirstChild(toolName)
    if dupedTool then
        dupedTool.Parent = character
    end

    if connection then
        connection:Disconnect()
    end

    task.wait(2)
    DupeCooldown = false
end

function FindMarketToolId(ToolName)
    if not ReplicatedStorage_KTD:FindFirstChild("MarketItems") then return nil end
    for _, Item in ipairs(ReplicatedStorage_KTD.MarketItems:GetChildren()) do
        if Item.Name == ToolName and Item:FindFirstChild("owner") and Item.owner.Value == LocalPlayer_KTD.Name then
            return Item:GetAttribute("SpecialId") or Item:GetAttribute("ToolId")
        end
    end
end

function MarketProtectTool(ToolName)
    if MarketProtectedTools[ToolName] then return end
    local ToolId
    local Connection
    if ReplicatedStorage_KTD:FindFirstChild("MarketItems") then
        Connection = ReplicatedStorage_KTD.MarketItems.ChildAdded:Connect(function(Item)
            if Item.Name == ToolName and Item:FindFirstChild("owner") and Item.owner.Value == LocalPlayer_KTD.Name then
                ToolId = Item:GetAttribute("SpecialId")
            end
        end)
    end
    pcall(function()
        if LocalPlayer_KTD.Character and LocalPlayer_KTD.Character:FindFirstChild("Humanoid") then
            LocalPlayer_KTD.Character:FindFirstChild("Humanoid"):UnequipTools()
        end
    end)
    task.wait(0.15)
    pcall(function()
        if ReplicatedStorage_KTD:FindFirstChild("ListWeaponRemote") then
            ReplicatedStorage_KTD.ListWeaponRemote:FireServer(ToolName, 676767)
        end
    end)
    task.wait(0.3)
    pcall(function()
        if ReplicatedStorage_KTD:FindFirstChild("BackpackRemote") then
            ReplicatedStorage_KTD.BackpackRemote:InvokeServer("Store", ToolName)
        end
    end)
    task.wait(3)
    if not ToolId then ToolId = FindMarketToolId(ToolName) end
    if Connection then Connection:Disconnect() end
    if ToolId then MarketProtectedTools[ToolName] = ToolId end
end

function MarketReleaseTool(ToolName, ToolId)
    if not ToolName or not ToolId then return end
    if not ReplicatedStorage_KTD:FindFirstChild("BuyItemRemote") or not ReplicatedStorage_KTD:FindFirstChild("BackpackRemote") then return end
    pcall(function() ReplicatedStorage_KTD.BuyItemRemote:FireServer(ToolName, "Remove", ToolId) end)
    task.wait(0.5)
    pcall(function() ReplicatedStorage_KTD.BackpackRemote:InvokeServer("Grab", ToolName) end)
    MarketProtectedTools[ToolName] = nil
end

function ProtectInventoryTools()
    for _, ToolName in ipairs((function()
        local names = {}
        if LocalPlayer_KTD.Character then
            local tool = LocalPlayer_KTD.Character:FindFirstChildOfClass("Tool")
            if tool then table.insert(names, tool.Name) end
        end
        return names
    end)()) do
        MarketProtectTool(ToolName)
    end
end

function ReleaseMarketTools()
    for ToolName, ToolId in pairs(MarketProtectedTools) do
        MarketReleaseTool(ToolName, ToolId)
    end
end

function SetupKeepTools(character)
    if not character then return end
    local humanoid = character:WaitForChild("Humanoid")
    humanoid.Died:Connect(function()
        if KeepToolsOnDeath then

        if not KeepToolsOnDeath then return end
        if value > 0 and value <= MarketHealthThreshold then
            task.spawn(ProtectInventoryTools)
        elseif value > MarketHealthThreshold and next(MarketProtectedTools) then
            task.spawn(ReleaseMarketTools)
        end
    end)
end

-- ==================== IMPROVED BRING NEAREST CAR ====================
function GetNearestCar(hrp)
    local CivCars = workspace:FindFirstChild("CivCars")
    if not CivCars then return nil end
    local nearestCar = nil
    local nearestDist = math.huge
    for _, car in ipairs(CivCars:GetChildren()) do
        if not car:IsA("Model") then continue end
        local seat = car:FindFirstChild("DriveSeat") or car:FindFirstChildWhichIsA("VehicleSeat")
        if not seat or seat.Occupant then continue end
        if not car.PrimaryPart then
            car.PrimaryPart = car:FindFirstChildWhichIsA("BasePart")
        end
        if not car.PrimaryPart then continue end
        local dist = (car.PrimaryPart.Position - hrp.Position).Magnitude
        if dist < nearestDist then
            nearestDist = dist
            nearestCar = car
        end
    end
    return nearestCar
end

-- Continuously apply all gun mods to equipped weapon
function applyGunMods()
    local char = LocalPlayer.Character
    if not char then return end
    local tool = char:FindFirstChildOfClass("Tool")
    if not tool or not tool:FindFirstChild("Setting") then return end

    pcall(function()
        local setting = require(tool.Setting)

        if GunMods.InfiniteAmmo then
            setting.Ammo = 99999
            setting.AmmoPerMag = 99999
            setting.MaxAmmo = 99999
            setting.LimitedAmmoEnabled = false
        end
        if GunMods.Bullets80k then
            setting.Ammo = 80000
            setting.AmmoPerMag = 80000
            setting.MaxAmmo = 80000
            setting.LimitedAmmoEnabled = false
            local gunLocal = tool:FindFirstChild("GunScript_Local")
            if gunLocal and type(getsenv) == "function" then
                pcall(function()
                    local env = getsenv(gunLocal)
                    if env and type(env.Reload) == "function" and type(debug.setupvalue) == "function" then
                        debug.setupvalue(env.Reload, 1, 80000)
                        debug.setupvalue(env.Reload, 3, 80000)
                    end
                end)
            end
        end
        if GunMods.InstantReload then
            setting.ReloadTime = 0
        end
        if GunMods.InstantEquip then
            setting.EquipTime = 0
            setting.EquippingTime = 0
        end
        if GunMods.Unjam then
            setting.JamChance = 0
        end
        if GunMods.NoRecoil then
            setting.Recoil = 0
            setting.CameraRecoilingEnabled = false
        end
        if GunMods.Automatic then
            setting.Auto = true
        end
        if GunMods.NoFireRate then
            setting.FireRate = 0
        end
        if GunMods.InfiniteDamage then
            setting.BaseDamage = 9e9
        end
        if GunMods.NoSpread then
            setting.Spread = 0
            setting.SpreadX = 0
            setting.SpreadY = 0
            setting.Accuracy = 1
        end
        if GunMods.Flamethrower then
            setting.FireRate = 0
            setting.Auto = true
            setting.Spread = 0
            setting.BaseDamage = math.max(setting.BaseDamage or 10, 15)
            -- Enable fire/ignite effect on the gun
            local gunServer = tool:FindFirstChild("GunScript_Server")
            if gunServer then
                local ignite = gunServer:FindFirstChild("IgniteScript")
                if ignite then
                    -- Force ignite on every shot
                    setting.Ignite = true
                    setting.IgniteScript = ignite
                    setting.IgniteChance = 1
                    setting.IgniteDuration = 5
                    setting.IgniteDamage = 10
                end
            end
            -- Make it shoot like a flamethrower (multiple bullets per shot)
            setting.ShotgunEnabled = true
            setting.Pellets = 12
            setting.SpreadX = 0.3
            setting.SpreadY = 0.3
        end
    end)
end

-- Helper: get good cleaner
function GetGoodCleaner()
    if not workspace:FindFirstChild("1# Map") then return nil end
    local CounterInstance
    for _, Value in pairs(workspace["1# Map"]:GetChildren()) do
        if Value:FindFirstChild("CounterM") then
            CounterInstance = Value
            break
        end
    end
    if not CounterInstance then return nil end
    for _, _Value in pairs(CounterInstance:GetChildren()) do
        local cashPrompt = _Value:FindFirstChild("CashPrompt", true)
        local grabPrompt = _Value:FindFirstChild("GrabPrompt", true)
        if cashPrompt and cashPrompt.Enabled and cashPrompt.ObjectText == "Count Bread" and grabPrompt and not grabPrompt.Enabled then
            return _Value
        end
    end
    return nil
end

-- Drop money amount
function DropMoneyAmount(Amount)
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local BankRemote = ReplicatedStorage:FindFirstChild('BankProcessRemote', true)
    if not BankRemote then return false end

function send(...)
        if BankRemote.InvokeServer then return BankRemote:InvokeServer(...)
        elseif BankRemote.FireServer then return BankRemote:FireServer(...) end
    end

    local attempts = {
        {"Drop", tostring(Amount)},
        {"Drop", tonumber(Amount) or Amount},
        {{"Drop", tostring(Amount)}},
        {{"Drop", tonumber(Amount) or Amount}},
    }
    for _, args in ipairs(attempts) do
        local ok = pcall(send, table.unpack(args))
        if ok then return true end
    end
    return false
end

-- Auto deposit
function autoDepositAmount(amount)
    pcall(function()
        local BankAction = game:GetService("ReplicatedStorage"):FindFirstChild("BankAction")
        if BankAction then
            if amount and tonumber(amount) and tonumber(amount) > 0 then
                BankAction:FireServer("depo", tostring(math.floor(amount)))
            else
                BankAction:FireServer("depo", tostring(999999999))
            end
        end
    end)
end

-- Format money
function formatMoney(amount)
    local formatted = tostring(math.floor(tonumber(amount) or 0))
    while true do
        local newFormatted, count = formatted:gsub("^(-?%d+)(%d%d%d)", "%1,%2")
        formatted = newFormatted
        if count == 0 then break end
    end
    return "$" .. formatted
end

function createDrawing(className, props)
    if not Drawing or type(Drawing.new) ~= "function" then
        return {Visible = false, Remove = function() end}
    end
    local ok, obj = pcall(Drawing.new, className)
    if not ok or not obj then return {Visible = false, Remove = function() end} end

    return obj

local SkeletonPairs = {
    {"Head", "UpperTorso"}, {"UpperTorso", "LowerTorso"},
    {"UpperTorso", "LeftUpperArm"}, {"LeftUpperArm", "LeftLowerArm"}, {"LeftLowerArm", "LeftHand"},
    {"UpperTorso", "RightUpperArm"}, {"RightUpperArm", "RightLowerArm"}, {"RightLowerArm", "RightHand"},
    {"LowerTorso", "LeftUpperLeg"}, {"LeftUpperLeg", "LeftLowerLeg"}, {"LeftLowerLeg", "LeftFoot"},
    {"LowerTorso", "RightUpperLeg"}, {"RightUpperLeg", "RightLowerLeg"}, {"RightLowerLeg", "RightFoot"},
}

function createESPObject(player)
    if player == Players.LocalPlayer or ESPObjects[player] then return end
    local box = {}
    for _, name in ipairs({"Top", "Bottom", "Left", "Right", "TopLeft", "TopRight", "BottomLeft", "BottomRight"}) do
        box[name] = createDrawing("Line", {Visible = false, Thickness = 1, Color = ESPFlags.BoxColor})
    end
    local skeleton = {}
    for i = 1, #SkeletonPairs do
        skeleton[i] = createDrawing("Line", {Visible = false, Thickness = 1, Color = ESPFlags.SkeletonColor})
    end
    ESPObjects[player] = {
        Box = box,
        Fill = createDrawing("Square", {Visible = false, Filled = true, Transparency = 0.85, Color = ESPFlags.BoxFillColor}),
        HealthOutline = createDrawing("Square", {Visible = false, Filled = false, Color = Color3.fromRGB(0,0,0), Thickness = 1}),
        HealthFill = createDrawing("Square", {Visible = false, Filled = true, Color = Color3.fromRGB(0,255,0)}),
        HealthText = createDrawing("Text", {Visible = false, Center = true, Outline = true, Size = ESPFlags.TextSize, Color = Color3.fromRGB(255,255,255)}),
        Name = createDrawing("Text", {Visible = false, Center = true, Outline = true, Size = ESPFlags.TextSize, Color = ESPFlags.TextColor}),
        Distance = createDrawing("Text", {Visible = false, Center = true, Outline = true, Size = ESPFlags.TextSize, Color = ESPFlags.TextColor}),
        Weapon = createDrawing("Text", {Visible = false, Center = true, Outline = true, Size = ESPFlags.TextSize, Color = ESPFlags.TextColor}),
        Snapline = createDrawing("Line", {Visible = false, Thickness = 1, Color = ESPFlags.SnaplineColor}),
        LookTracer = createDrawing("Line", {Visible = false, Thickness = 1, Color = ESPFlags.LookTracerColor}),
        Skeleton = skeleton,
    }
end

function hideESPObject(obj)
    if not obj then return end
    if obj.Box then for _, line in pairs(obj.Box) do line.Visible = false end end
    if obj.Skeleton then for _, line in ipairs(obj.Skeleton) do line.Visible = false end end
    for _, key in ipairs({"Fill", "HealthOutline", "HealthFill", "HealthText", "Name", "Distance", "Weapon", "Snapline", "LookTracer"}) do
        if obj[key] then obj[key].Visible = false end
    end
end

function removeESPObject(player)
    local obj = ESPObjects[player]
    if obj then
        if obj.Box then for _, line in pairs(obj.Box) do pcall(function() line:Remove() end) end end
        if obj.Skeleton then for _, line in ipairs(obj.Skeleton) do pcall(function() line:Remove() end) end end
        for _, key in ipairs({"Fill", "HealthOutline", "HealthFill", "HealthText", "Name", "Distance", "Weapon", "Snapline", "LookTracer"}) do
            if obj[key] then pcall(function() obj[key]:Remove() end) end
        end
        ESPObjects[player] = nil
    end
    if HighlightObjects[player] then
        pcall(function() HighlightObjects[player]:Destroy() end)
        HighlightObjects[player] = nil
    end
end

function getESPColor(player, defaultColor)
    if ESPFlags.TeamColor and player.Team and Players.LocalPlayer.Team and player.Team == Players.LocalPlayer.Team then
        return Color3.fromRGB(0,120,255)
    end
    return defaultColor
end

function updateLine(line, from, to, color)
    if not line then return end
    line.From = from
    line.To = to
    line.Color = color
    line.Visible = true
end

    local camera = workspace.CurrentCamera
    if not camera then return end
    for _, player in ipairs(Players:GetPlayers()) do
        if player == Players.LocalPlayer then continue end
        createESPObject(player)
        local obj = ESPObjects[player]
        local character = player.Character
        local humanoid = character and character:FindFirstChildOfClass("Humanoid")

            continue

        local distance = localRoot and (root.Position - localRoot.Position).Magnitude or (root.Position - camera.CFrame.Position).Magnitude
        if distance > ESPFlags.RenderDistance then
            hideESPObject(obj)
            if HighlightObjects[player] then HighlightObjects[player].Enabled = false end
            continue

        local rootPos, onScreen = camera:WorldToViewportPoint(root.Position)

                updateLine(obj.Box.Right, tr, tr + Vector2.new(0, corner), boxColor)

                updateLine(obj.Box.Bottom, bl, bl - Vector2.new(0, corner), boxColor)

            else obj.LookTracer.Visible = false end

        if ESPFlags.Skeleton then
            for i, pair in ipairs(SkeletonPairs) do
                local a, b = character:FindFirstChild(pair[1]), character:FindFirstChild(pair[2])
                if a and b and obj.Skeleton[i] then
                    local pa, oa = camera:WorldToViewportPoint(a.Position)

                        obj.Skeleton[i].Color = ESPFlags.SkeletonColor
                        obj.Skeleton[i].Visible = true
                    else obj.Skeleton[i].Visible = false end
                elseif obj.Skeleton[i] then obj.Skeleton[i].Visible = false end
            end
        else
            for _, line in ipairs(obj.Skeleton) do line.Visible = false end
        end
    end
end)

-- Get items from InvData (safe contents)
function GetSafeItems()
    local items = {}
    local invData = LocalPlayer:FindFirstChild("InvData")
    if invData then
        for _, child in ipairs(invData:GetChildren()) do
            table.insert(items, child.Name)
        end
    end
    table.sort(items)
    return items
end

-- Find safe folder
function GetSafeFolder()
    return workspace:FindFirstChild("HomeDoorSS") 
        and workspace.HomeDoorSS:FindFirstChild("HomeDoor") 
        and workspace.HomeDoorSS.HomeDoor:FindFirstChild("Safe")
end

-- Take item from safe
function TakeFromSafe(itemName)
    local SafeFolder = GetSafeFolder()
    if not SafeFolder then
        _fixNotify("Safe", "Safe not found!", 3)
        return false
    end
    
    local safePart = SafeFolder:FindFirstChild("Union") or SafeFolder:FindFirstChildWhichIsA("BasePart")
    if not safePart then return false end
    
    local char = LocalPlayer.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return false end
    
    local origLoc = root.CFrame
    local success = false
    
    pcall(function()
        _fixNotify("Safe", "Taking: " .. itemName, 2)
        Config:Teleport(safePart.CFrame * CFrame.new(0, 2, 0))
        task.wait(0.3)
        ReplicatedStorage:FindFirstChild("Inventory"):FireServer("Change", itemName, "Inv", SafeFolder)
        task.wait(0.5)
        Config:Teleport(origLoc)
        task.wait(0.5)
    end)
    
    local got = LocalPlayer.Backpack:FindFirstChild(itemName) 
        or (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild(itemName))
    
    if got then
        _fixNotify("Safe", "Got: " .. itemName, 3)
        success = true
    else
        _fixNotify("Safe", "Failed: " .. itemName, 3)
    end
    
    return success
end

-- Store item to safe
function StoreToSafe(itemName)
    local SafeFolder = GetSafeFolder()
    if not SafeFolder then
        _fixNotify("Safe", "Safe not found!", 3)
        return false
    end
    
    local safePart = SafeFolder:FindFirstChild("Union") or SafeFolder:FindFirstChildWhichIsA("BasePart")
    if not safePart then return false end
    
    local char = LocalPlayer.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return false end
    
    local origLoc = root.CFrame
    
    -- Unequip first
    local hum = char:FindFirstChildOfClass("Humanoid")
    if hum then hum:UnequipTools() end
    task.wait(0.2)
    
    pcall(function()
        _fixNotify("Safe", "Storing: " .. itemName, 2)
        Config:Teleport(safePart.CFrame * CFrame.new(0, 2, 0))
        task.wait(0.3)
        ReplicatedStorage:FindFirstChild("BackpackRemote"):InvokeServer("Store", itemName)
        ReplicatedStorage:FindFirstChild("Inventory"):FireServer("Change", itemName, "Backpack", SafeFolder)
        task.wait(0.5)
        Config:Teleport(origLoc)
        task.wait(0.5)
    end)
    
    local stillHave = LocalPlayer.Backpack:FindFirstChild(itemName) 
        or (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild(itemName))
    
    if not stillHave then
        _fixNotify("Safe", "Stored: " .. itemName, 3)
        return true
    else
        _fixNotify("Safe", "Store failed: " .. itemName, 3)
        return false
    end
end

function MarketHasItem(toolName)
        local market = ReplicatedStorage:FindFirstChild("MarketItems")
        if not market then return false end
        for _, item in ipairs(market:GetChildren()) do
            if item.Name == toolName and item:FindFirstChild("owner") and item.owner.Value == LocalPlayer.Name then

            end
        end
        return false
    end

function ForceRetrieveFromMarket(toolName)
        local market = ReplicatedStorage:WaitForChild("MarketItems", 10)

        local endTime = os.clock() + 12
        while os.clock() < endTime do
            for _, item in ipairs(market:GetChildren()) do

                    pcall(function() BuyItemRemote:FireServer(toolName, "Remove", item:GetAttribute("SpecialId")) end)

                    task.wait(0.45)
                    if LocalPlayer.Backpack:FindFirstChild(toolName) then return true end
                end
            end
            task.wait(0.2)
        end
        return false
    end

function getPing()
        local ok, result = pcall(function()
            return tonumber(string.match(game.Players.LocalPlayer:GetNetworkPing(), "%d+"))
        end)
        if ok and result then return result end
        return 100
    end

function duplicateCurrentItem()
        if DupeCooldown then
            _fixNotify("Dupe", "Cooldown! Please wait.")
            return
        end
        DupeCooldown = true

        local lp = game.Players.LocalPlayer
        local character = lp.Character
        local backpack = lp:FindFirstChild("Backpack")
        if not character or not backpack then DupeCooldown = false; return end

        local tool = character:FindFirstChildOfClass("Tool")
        if not tool then _fixNotify("Dupe", "Hold a tool first!"); DupeCooldown = false; return end

        local toolName = tool.Name
        local toolId = nil

        tool.Parent = backpack
        task.wait(0.5)

        local connection
        if RS:FindFirstChild("MarketItems") then
            connection = RS.MarketItems.ChildAdded:Connect(function(item)
                if item.Name == toolName then
                    local owner = item:WaitForChild("owner", 2)
                    if owner and owner.Value == lp.Name then
                        toolId = item:GetAttribute("SpecialId")
                    end
                end
            end)
        end

        task.spawn(function()
            if RS:FindFirstChild("ListWeaponRemote") then RS.ListWeaponRemote:FireServer(toolName, 676767) end
        end)

        local ping = getPing()
        task.wait(0.25 + ((math.clamp(ping, 0, 300) / 300) * 0.03))

        task.spawn(function()
            if RS:FindFirstChild("BackpackRemote") then RS.BackpackRemote:InvokeServer("Store", toolName) end
        end)

        task.wait(3)

        if toolId then
            task.spawn(function()
                if RS:FindFirstChild("BuyItemRemote") then RS.BuyItemRemote:FireServer(toolName, "Remove", toolId) end
            end)
        end

        task.spawn(function()
            if RS:FindFirstChild("BackpackRemote") then RS.BackpackRemote:InvokeServer("Grab", toolName) end
        end)

        task.wait(1)

        local dupedTool = backpack:FindFirstChild(toolName)
        if dupedTool and LocalPlayer.Character then dupedTool.Parent = LocalPlayer.Character end
        if connection then connection:Disconnect() end
        task.wait(2)
        DupeCooldown = false
    end

    task.spawn(function()
        while task.wait(0.5) do
            if AutoDupeEnabled and not DupeCooldown then
                local tool = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Tool")
                if tool then DupeToolName = tool.Name; pcall(duplicateCurrentItem) end
            end
        end
    end)

function GetBackpackTools()

        local bp = LocalPlayer:FindFirstChild("Backpack")

        if char then
            for _, t in pairs(char:GetChildren()) do
                if t:IsA("Tool") and not table.find(items, t.Name) then table.insert(items, t.Name) end
            end
        end
        if bp then
            for _, t in pairs(bp:GetChildren()) do
                if t:IsA("Tool") and not table.find(items, t.Name) then table.insert(items, t.Name) end
            end
        end
        table.sort(items)
        return items
    end
